## 2. URL Structure

```
/en/products/nike-air-max          (English, LTR)
/fa/products/nike-air-max          (Persian, RTL)
```

### Routing Strategy

```typescript
// app/[locale]/layout.tsx
export function generateStaticParams() {
  return [{ locale: 'en' }, { locale: 'fa' }];
}

// Middleware for locale detection
export function middleware(request: NextRequest) {
  const pathname = request.nextUrl.pathname;
  
  // Check if locale is in URL
  const pathnameHasLocale = locales.some(
    (locale) => pathname.startsWith(`/${locale}/`) || pathname === `/${locale}`
  );
  
  if (pathnameHasLocale) return;
  
  // Detect preferred locale
  const acceptLanguage = request.headers.get('accept-language');
  const preferredLocale = detectLocale(acceptLanguage);
  
  // Redirect to localized URL
  request.nextUrl.pathname = `/${preferredLocale}${pathname}`;
  return Response.redirect(request.nextUrl);
}
```

## 4. RTL Support

### CSS Strategy

```css
/* globals.css */
:root {
  --direction: ltr;

  --direction: rtl;
}

.start { text-align: start; }
.end { text-align: end; }
.margin-start { margin-inline-start: 1rem; }
.margin-end { margin-inline-end: 1rem; }
.padding-start { padding-inline-start: 1rem; }
.padding-end { padding-inline-end: 1rem; }
.border-start { border-inline-start: 1px solid; }
.border-end { border-inline-end: 1px solid; }

/* Flexbox & Grid */
.flex-row { display: flex; flex-direction: row; }
[data-direction="rtl"] .flex-row { flex-direction: row-reverse; }

/* Transform for icons */
[data-direction="rtl"] .flip-rtl { transform: scaleX(-1); }
```

### Component Pattern

```typescript
// lib/i18n.ts
export function getDirection(locale: string): 'ltr' | 'rtl' {
  return locale === 'fa' ? 'rtl' : 'ltr';
}

// app/[locale]/layout.tsx
export default function RootLayout({ children, params }) {
  const direction = getDirection(params.locale);
  
  return (
    <html lang={params.locale} dir={direction}>
      <body>{children}</body>
    </html>
  );
}
```

## 6. Currency Localization

```typescript
const currencyConfig = {
  USD: { symbol: '$', position: 'before', decimal: 2, thousands: ',' },
  IRR: { symbol: '﷼', position: 'after', decimal: 0, thousands: ',' },
  EUR: { symbol: '€', position: 'before', decimal: 2, thousands: '.' },
};

export function formatPrice(amount: number, currency: string, locale: string): string {
  const config = currencyConfig[currency];
  
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency,
    minimumFractionDigits: config.decimal,
    maximumFractionDigits: config.decimal,
  }).format(amount);
}

// Usage
formatPrice(1234.56, 'USD', 'en');  // $1,234.56
formatPrice(12345678, 'IRR', 'fa');  // ۱۲,۳۴۵,۶۷۸ ریال
```

## 7. Date/Time Localization

```typescript
export function formatDate(date: Date, locale: string): string {
  return new Intl.DateTimeFormat(locale, {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
  }).format(date);
}

export function formatDateTime(date: Date, locale: string): string {
  return new Intl.DateTimeFormat(locale, {
    year: 'numeric',
    month: 'short',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
  }).format(date);
}

// Persian calendar conversion (future)
export function toPersianDate(date: Date): string {
  // Convert to Jalali calendar
  return new Intl.DateTimeFormat('fa-IR', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
  }).format(date);
}
```

  type: 'product' | 'category' | 'brand',
  slug: string,
  locale: string,
): string {

## 9. Translation Workflow

### For Developers

1. Add translation keys to `/messages/en/*.json`
2. Run `pnpm i18n:extract` to update translation files
3. Send new keys to translators
4. Import translated files to `/messages/fa/*.json`

### For Translators

1. Use translation management platform (Crowdin, Lokalise)
2. Translate context-aware strings
3. Review machine translations
4. Approve final translations

### Automated Checks

```typescript
// scripts/check-translations.ts
async function checkMissingTranslations() {
  const enFiles = await glob('/messages/en/*.json');
  const faFiles = await glob('/messages/fa/*.json');
  
  for (const enFile of enFiles) {
    const faFile = enFile.replace('/en/', '/fa/');
    
    if (!fs.existsSync(faFile)) {
      console.error(`Missing translation file: ${faFile}`);
      continue;
    }
    
    const enKeys = flattenKeys(JSON.parse(fs.readFileSync(enFile, 'utf-8')));
    const faKeys = flattenKeys(JSON.parse(fs.readFileSync(faFile, 'utf-8')));
    
    const missing = Object.keys(enKeys).filter((key) => !(key in faKeys));
    
    if (missing.length > 0) {
      console.error(`Missing translations in ${faFile}:`, missing);
    }
  }
}
```
